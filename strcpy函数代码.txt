#include <stdio.h>
#include<stdlib.h>
int getLength(char str[]){
    char *p=str;
    int count=0;
    while(*p++ !='\0'){
        count++;
    }
    return count;

}
char* strcpy(char src[]){
    int length= getLength( src);
    char *dst=(char*)malloc(length+1);
    for(int i=0;i<=length;i++){
        dst[i]=src[i];
    }
    dst[length+1]='\0';
    return dst;
}

    printf("%s",strcpy("asdf"));
